    def get_closest(tuple_a: tuple, tuple_list: list) -> tuple:
    """Return closest tuple based on first coordinate criteria"""
    least = 0
    dist = tuple_a[0] - tuple_list[0][0]
    for i in range(len(tuple_list)):
        if not tuple_list[i][0] <= tuple_a[0]:
            continue
        n_dist = tuple_a[0] - tuple_list[i][0]
        if n_dist <= dist:
            dist = n_dist
            least = i
    return tuple_list[least]

    def get_new_spans(data: str, sections: dict) -> dict:
    """Realign spans of sections after Masking

    Parameters
    ----------
    data: str
        Text information
    sections: dict
        Dictionary with sections and their spans
    """

    spans = []
    for match in re.finditer(r"\[SECTION\]", data):
        spans.append(match.span())

    for (start, end) in spans:
        assert end == start + 9, f"Span is of length {end - start}"

    for section, sec_spans in sections.items():
        n_spans = []
        for span in sec_spans:
            n_span = get_closest(span, spans)
            n_spans.append(n_span)
        sections[section] = n_spans

    return sections

    def mask_statutes(data: str, sections: dict) -> Union[str, dict]:
    """Mask given sections from sentences and remove certain elements

    Parameters
    ----------
    data: str
        Text
    sections: dict
        Dictionary of all sections cited in document along with their spans
    """
    data_copy = deepcopy(data)
    if sections:

        alt_case_section_regex = re.compile(
                "|".join([r"Section\s+{}\s+of\s+the\s+(Code|Act)".format(
                    sec.split("_")[-1])
                    for sec in sections]))
        replace_texts = []
        for sec, spans in sections.items():
            replace_texts.extend([data_copy[start:end]
                                  for (start, end) in spans])
        replace_texts = [rep.replace("(", r"\(").replace(")", r"\)")
                         for rep in replace_texts]

        data_copy = re.sub("|".join(replace_texts), "[SECTION]", data_copy)
        data_copy = alt_case_section_regex.sub("[SECTION_UNK]", data_copy)

    alt_case_section_regex_2 = re.compile(r"Sections?\s+(?:[0-9]+[A-Z]*)")
    data_copy = alt_case_section_regex_2.sub("[SECTION_UNK]", data_copy)
    n_spans = get_new_spans(data_copy, sections)

    return data_copy, n_spans
